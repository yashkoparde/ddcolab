4a: Half Adder

//4a
// Design Code
module halfadder (
    input  a,
    input  b,
    output sum,
    output carry
);

    assign sum   = a ^ b;
    assign carry = a & b;

endmodule
// Testbench

    reg b;

    wire carry;

    // Instantiate the Half Adder
    halfadder uut (
        .a(a),
        .b(b),
        .sum(sum),
        .carry(carry)
    );

    // Apply test vectors

        a = 0; b = 0; #100;
        a = 0; b = 1; #100;
        a = 1; b = 0; #100;

    end

    // Monitor outputs

        $monitor("Time=%0t | a=%b | b=%b | sum=%b | carry=%b",
                  $time, a, b, sum, carry);
    end

endmodule

5: BCD Adder

//5
// Code your design here
module bcd_adder (
    input  [3:0] A,
    input  [3:0] B,
    input        Cin,
    output reg [3:0] Sum,
    output reg       Cout
);

    reg [4:0] temp_sum; // 5-bit temporary sum

    always @(A or B or Cin) begin
        temp_sum = A + B + Cin;   // Binary addition

        if (temp_sum > 9) begin
            Sum  = temp_sum[3:0] + 4'b0110; // BCD correction
            Cout = 1'b1;
        end
        else begin
            Sum  = temp_sum[3:0];
            Cout = 1'b0;
        end
    end

endmodule

// Code your testbench here

// Code your design here
module mux_2x1_behavioral (
    input  I0,    // Data input 0
    input  I1,    // Data input 1
    input  S,     // Select input
    output reg Y  // Output
);

    always @(*) begin
        if (S == 1'b0)
            Y = I0;
        else
            Y = I1;
    end

endmodule

7a: 1-to-2 Demultiplexer

//7a
// 1-to-2 Demultiplexer
module demux1x2 (
    input  wire din,   // Data input
    input  wire sel,   // Select line
    output reg y0,     // Output 0
    output reg y1      // Output 1
);

    always @(*) begin
        // Default outputs
        y0 = 0;
        y1 = 0;

        // Route input to selected output
        if (sel == 0)
            y0 = din;
        else
            y1 = din;
    end

endmodule
// Testbench for 1-to-2 DEMUX

    // Outputs

        .din(din),
        .sel(sel),
        .y0(y0),
        .y1(y1)
    );

    // Simulation
    initial begin
        $display("Time   din sel | y0 y1");
        $monitor("%0t     %b   %b   | %b  %b", $time, din, sel, y0, y1);

        // Test case 1: din = 1
        din = 1;
        sel = 0; #10;   // y0 = 1

        // Test case 2: din = 0
        din = 0;
        sel = 0; #10;   // y0 = 0
        sel = 1; #10;   // y1 = 0

        $finish;
    end

endmodule

8a:SR Flip-Flop

//8a
// SR Flip-Flop Design
module SR_FF (
    input S,          
    input R,          
    input clk,        
    output reg Q      
);

    always @(posedge clk) begin
        case ({S, R})
            2'b10: Q <= 1;    
            2'b01: Q <= 0;    
            2'b00: Q <= Q;    
            2'b11: Q <= 1'bx; 
        endcase
    end

endmodule
// Testbench for SR Flip-Flop
module TB_SR_FF();
    // Inputs
    reg S, R, clk;

    // Output
    wire Q;

    // Instantiate the SR Flip-Flop
    SR_FF uut (
        .S(S),
        .R(R),
        .clk(clk),
        .Q(Q)
    );

    // Clock generation: 10-time-unit period
    initial begin
        clk = 0;
        forever #5 clk = ~clk;
    end

    // Stimulus and monitoring
    initial begin
        $monitor("Time=%0t | S=%b R=%b Q=%b", $time, S, R, Q);

        // Test sequence
        S = 0; R = 0; #10;  
        S = 1; R = 0; #10;  
        S = 0; R = 1; #10;  
        S = 0; R = 0; #10;  
        S = 1; R = 1; #10;  

        $stop; // Stop simulation
    end
endmodule

//8b
// JK Flip-Flop (Clocked) Design
module JK_FF (
    input J,         
    input K,          
    input clk,       
    output reg Q      
);

    always @(posedge clk) begin
        case ({J, K})
            2'b00: Q <= Q;     
            2'b01: Q <= 0;     
            2'b10: Q <= 1;     
            2'b11: Q <= ~Q;    
        endcase
    end

endmodule
// Testbench for JK Flip-Flop
